68 void buildInodeProcessMap()
69 {
70 auto proc_dir = opendir("/proc");
71 if (!proc_dir) return;
72
73 struct dirent* entry;
74 while ((entry = readdir(proc_dir)) != nullptr)
75 {
76 std::string dirname = entry->d_name;
77
78 if (std::all_of(dirname.begin(), dirname.end(), ::isdigit))
79 {
80 pid_t pid = std::stoi(dirname);
81 std::string fd_path = "/proc/" + dirname + "/fd/";
82
83 std::string comm_path = "/proc/" + dirname + "/comm";
84 std::ifstream comm_file(comm_path);
85 std::string process_name;
86 if (comm_file)
87 std::getline(comm_file, process_name);
88
89 auto fd_dir = opendir(fd_path.c_str());
90 if (fd_dir)
91 {
92 struct dirent* fd_entry;
93 while ((fd_entry = readdir(fd_dir)) != nullptr)
94 {
95 std::string fd_name = fd_entry->d_name;
96 if (fd_name == "." || fd_name == "..") continue;
97
98 std::string link_path = fd_path + fd_name;
99 char link_target[256];
100 ssize_t len = readlink(link_path.c_str(), link_target, sizeof(link_target)-1);
101 if (len != -1)
102 {
103 link_target[len] = '\0';
104 std::string target_str(link_target);
105
106 size_t pos = target_str.find("socket:[");
107 if (pos != std::string::npos)
108 {
109 std::string inode = target_str.substr(pos + 8);
110 inode.pop_back();
111 inode_to_process[inode] = {pid, process_name};
112 }
113 }
114 }
115 closedir(fd_dir);
116 }
117 }
118 }
119 closedir(proc_dir);

122 std::string hexToIp(const std::string& hex_ip)
123 {
124 if (hex_ip.length() != 8) return "0.0.0.0";
125
126 unsigned int ip_int;
127 std::stringstream ss;
128 ss << std::hex << hex_ip;
129 ss >> ip_int;
130
131 return std::to_string(ip_int & 0xFF) + "." +
132 std::to_string((ip_int >> 8) & 0xFF) + "." +
133 std::to_string((ip_int >> 16) & 0xFF) + "." +
134 std::to_string((ip_int >> 24) & 0xFF);
135 }

151 std::vector<ConnectionInfo> scanConnections()
152 {
153 std::vector<ConnectionInfo> connections;
154
155 std::vector<std::pair<std::string, std::string>> proc_files = {
156 {"/proc/net/tcp", "TCP"},
157 {"/proc/net/udp", "UDP"},
158 {"/proc/net/tcp6", "TCP6"},
159 {"/proc/net/udp6", "UDP6"}
160 };
161
162 for (const auto& [filename, protocol] : proc_files)
163 {
164 std::ifstream file(filename);
165 if (!file.is_open()) continue;
166
167 std::string line;
168 std::getline(file, line);
169
170 while (std::getline(file, line)) {
171 std::istringstream iss(line);
172 std::vector<std::string> tokens;
173 std::string token;
174
175 while (iss >> token) {
176 tokens.push_back(token);
177 }
178
179 if (tokens.size() < 10) continue;
180
181 ConnectionInfo info;
182 info.protocol = protocol;
183
184 std::string local_addr = tokens[1];
185 size_t colon_pos = local_addr.find(':');
186 std::string local_ip_hex = local_addr.substr(0, colon_pos);
187 std::string local_port_hex = local_addr.substr(colon_pos + 1);
188
189 info.local_address = hexToIp(local_ip_hex);
190 info.local_port = hexToPort(local_port_hex);
191
192 std::string remote_addr = tokens[2];
193 colon_pos = remote_addr.find(':');
194 std::string remote_ip_hex = remote_addr.substr(0, colon_pos);
195 std::string remote_port_hex = remote_addr.substr(colon_pos + 1);
196
197 info.remote_address = hexToIp(remote_ip_hex);
198 info.remote_port = hexToPort(remote_port_hex);
199
200 std::string state_hex = tokens[3];
201 info.state = socket_states.count(state_hex) ?
202 socket_states[state_hex] : "UNKNOWN";
203
204 std::string inode = tokens[9];
205 if (inode_to_process.count(inode))
206 {
207 info.pid = inode_to_process[inode].first;
208 info.process_name = inode_to_process[inode].second;
209 }
210 else
211 {
212 info.pid = -1;
213 info.process_name = "unknown";
214 }
215
216 connections.push_back(info);
217 }
218
219 file.close();
220 }
221
222 return connections;
223 }
