76 void write(const std::string& level, const std::string& message)
77 {
78 std::lock_guard<std::mutex> lock(log_mutex_);
79
80 auto now = std::chrono::system_clock::now();
81 auto time_t = std::chrono::system_clock::to_time_t(now);
82 auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
83 now.time_since_epoch()) % 1000;
84
85 std::stringstream ss;
86 ss << std::put_time(std::localtime(&time_t), "%Y-%m-%d %H:%M:%S");
87 ss << "." << std::setfill('0') << std::setw(3) << ms.count();
88 std::string timestamp = ss.str();
89
90 std::string log_message = "[" + timestamp + "] [" + level + "] " + message;
91
92 if (log_to_console_)
93 {
94 if (level == "ERROR")
95 std::cerr << log_message << std::endl;
96 else
97 std::cout << log_message << std::endl;
98 }
99
100 if (log_to_file_ && log_file_.is_open())
101 {
102 log_file_ << log_message << std::endl;
103 log_file_.flush();
104 }
105 }
