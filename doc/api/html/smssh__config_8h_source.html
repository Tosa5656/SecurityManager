58 bool load() {
59 if (!fs::exists(config_path_)) {
60 save();
61 return false;
62 }
63
64 std::ifstream file(config_path_);
65 if (!file.is_open()) {
66 return false;
67 }
68
69 std::string line;
70 while (std::getline(file, line)) {
71 // Пропустить комментарии и пустые строки
72 if (line.empty() || line[0] == '#') {
73 continue;
74 }
75
76 size_t pos = line.find('=');
77 if (pos != std::string::npos) {
78 std::string key = line.substr(0, pos);
79 std::string value = line.substr(pos + 1);
80
81 // Обрезать пробелы
82 key.erase(0, key.find_first_not_of(" \t"));
83 key.erase(key.find_last_not_of(" \t") + 1);
84 value.erase(0, value.find_first_not_of(" \t"));
85 value.erase(value.find_last_not_of(" \t") + 1);
86
87 config_[key] = value;
88 }
89 }
90
91 file.close();
92 return true;
93 }

95 bool save() {
96 // Создать директорию если она не существует
97 fs::path config_dir = fs::path(config_path_).parent_path();
98 if (!config_dir.empty() && !fs::exists(config_dir)) {
99 fs::create_directories(config_dir);
100 }
101
102 std::ofstream file(config_path_);
103 if (!file.is_open()) {
104 return false;
105 }
106
107 file << "# SSH Security Manager Configuration\n";
108 file << "# Generated automatically\n\n";
109
110 for (const auto& [key, value] : config_) {
111 file << key << " = " << value << "\n";
112 }
113
114 file.close();
115 return true;
116 }
