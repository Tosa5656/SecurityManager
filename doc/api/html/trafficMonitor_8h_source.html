64 auto format_bytes = [](uint64_t bytes) -> std::string
65 {
66 if (bytes < 1024) return std::to_string(bytes) + " B";
67 double value = bytes;
68 const char* units[] = {"KB", "MB", "GB"};
69 int unit = 0;
70 while (value >= 1024 && unit < 2) {
71 value /= 1024;
72 unit++;
73 }
74 std::stringstream ss;
75 ss << std::fixed << std::setprecision(2) << value << " " << units[unit];
76 return ss.str();
77 };

110 struct pcap_pkthdr* header;
111 const u_char* packet_data;

113 uint64_t packets_this_sec = 0;

116 while (g_running.load())

131 switch (ip_header->ip_p)

133 case IPPROTO_TCP: stats.tcp_packets++; break;
134 case IPPROTO_UDP: stats.udp_packets++; break;
135 case IPPROTO_ICMP: stats.icmp_packets++; break;

137 }
138

142 stats.packets_last_sec = packets_this_sec;
143 stats.bytes_last_sec = bytes_this_sec;
144 packets_this_sec = 0;
145 bytes_this_sec = 0;
